def peritumoral_label(file_path,size = (7,7)):
element = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, size) #矩形：MORPH_RECT 交叉形：MORPH_CROSS; 椭圆形：MORPH_ELLIPSE;
for index,filepath in enumerate(glob.glob(os.path.join(file_path,’*’,’_seg.nii.gz’))):
nii_image = nib.load(filepath)
nii_data = nii_image.get_data()
nii_data[np.where(nii_data == 2)] = 0
nii_data[np.where(nii_data == 4)] = 1
images = np.zeros(nii_image.shape, dtype=np.uint8)
print(os.path.basename(filepath))
for slice in range(nii_data.shape[-1]):
slice_array = np.asanyarray(nii_data[:, :, slice], dtype=np.uint8)
slice_dilated = cv2.dilate(slice_array, element)
images[:,:,slice] = slice_dilated

nii = images - nii_data
new_nii = nib.Nifti1Image(nii, affine=nii_image.affine)
dirname = os.path.dirname(filepath)
nib.save(new_nii, os.path.join(dirname, '{}_seg_dilated_{}.nii.gz'.format(os.path.basename(dirname),int(size[0])-1)))
